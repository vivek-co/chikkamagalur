write a prog to find the factorial for the set of values.
#!/bin/bash
num="2 4 5"
for n in $num
do 
fact = 1
temp=$n
while [ $n -gt 0 ]
do
fact= `expr $n \*fact`
n=`expr $n-1`
done
echo " the factorial of the number $fact is $temp "
done

3* - 3rd * represents date (1-31)

4* - 4th * represents month (1-12)

5* - 5th * represents day (Monday to sunday)
